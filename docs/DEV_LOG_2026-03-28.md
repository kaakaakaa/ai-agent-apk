### 1. LlmClient.kt - 生产级改造

**文件**: `app/src/main/java/com/agent/apk/agent/cloud/LlmClient.kt`

**问题**: 原版没有重试机制、错误处理粗糙、无日志调试困难

**优化内容**:

```kotlin
// 1. 重试机制 + 指数退避
// - 针对 429 (RateLimit) 和 503 (Server Error)
// - 初始延迟 1 秒，每次翻倍，最多 3 次
// - 认证错误 (401) 不重试，直接抛异常

// 2. 连接池优化（单例模式）
@Volatile
private var sharedClient: OkHttpClient? = null

private fun createOptimizedClient(): OkHttpClient {
    return sharedClient ?: synchronized(this) {
        sharedClient ?: OkHttpClient.Builder().apply {
            connectionPool(okhttp3.ConnectionPool(
                maxIdleConnections = 10,
                keepAliveDuration = 5L,
                TimeUnit.MINUTES
            ))
            connectTimeout(30, TimeUnit.SECONDS)
            readTimeout(60, TimeUnit.SECONDS)
            writeTimeout(30, TimeUnit.SECONDS)
            retryOnConnectionFailure(true)

            // 日志拦截器
            addInterceptor(HttpLoggingInterceptor {
                Log.d(TAG, "HTTP: $message")
            }.apply { level = HttpLoggingInterceptor.Level.BODY })

            // 错误处理拦截器
            addInterceptor { chain ->
                val response = chain.proceed(chain.request())
                when (response.code) {
                    401 -> throw AuthenticationException("API Key 无效")
                    403 -> throw PermissionException("无权限")
                    429 -> throw RateLimitException("请求太频繁")
                    500, 502, 503, 504 -> throw ServerException("服务器错误")
                }
                response
            }
        }.build().also { sharedClient = it }
    }
}

// 3. 特定错误类型（便于 UI 层精准提示）
open class ApiException(message: String, val errorCode: String? = null) : Exception(message)
class AuthenticationException(message: String) : ApiException(message, "AUTH_ERROR")
class PermissionException(message: String) : ApiException(message, "PERMISSION_ERROR")
class RateLimitException(message: String) : ApiException(message, "RATE_LIMIT_ERROR")
class ServerException(message: String) : ApiException(message, "SERVER_ERROR")
class NetworkException(message: String) : ApiException(message, "NETWORK_ERROR")
class TimeoutException(message: String) : ApiException(message, "TIMEOUT_ERROR")
```

```kotlin
// 任务执行开始
suspend fun executeTask(userGoal: String) {
    Log.d(TAG, "Starting task execution for: $userGoal")
    val session = cloudAgent.startTask(userGoal)
    _sessionState.value = session

    var stepsExecuted = 0

    while (stepsExecuted < MAX_STEPS && session.status == SessionStatus.ACTIVE) {
        stepsExecuted++
        Log.d(TAG, "Executing step $stepsExecuted/$MAX_STEPS")

        try {
            // 获取 UI 树
            val uiTree = AccessibilityScanner.getUiTreeSync()

            // 执行 ReAct 单步
            val result = cloudAgent.step(session, uiTree?.toJson())
            _currentStep.value = ReActStep(...)

            if (result.isComplete) {
                Log.d(TAG, "Task completed at step $stepsExecuted")
                cloudAgent.endTask(session)
                return
            }

            // 执行动作 - 详细日志
            result.action?.let { action ->
                Log.d(TAG, ">>> Executing action: ${action::class.java.simpleName}")
                Log.d(TAG, ">>> Action details: $action")

                val startTime = System.currentTimeMillis()
                val success = executeAction(action)
                val endTime = System.currentTimeMillis()

                Log.d(TAG, "<<< Action completed in ${endTime - startTime}ms, success: $success")

                // 记录结果
                val lastRecord = session.executedActions.lastOrNull()
                if (lastRecord != null) {
                    lastRecord.result = ActionResult(
                        success = success,
                        message = if (success) "执行成功" else "执行失败：${action::class.java.simpleName}",
                        newUiTree = AccessibilityScanner.getUiTreeSync()
                    )
                    Log.d(TAG, "<<< ActionResult recorded: success=$success")
                } else {
                    Log.w(TAG, "<<< WARNING: No action record found to update!")
                }
            }
        } catch (e: Exception) {
            Log.e(TAG, "Error at step $stepsExecuted: ${e.message}", e)
            session.status = SessionStatus.FAILED
            return
        }
    }
}

// 点击动作详细日志
private suspend fun executeClick(action: ClickAction): Boolean {
    Log.d(TAG, "[Click] Starting click execution")
    return try {
        if (action.nodeId != null) {
            Log.d(TAG, "[Click] Attempting node click with nodeId: ${action.nodeId}")
            val node = AccessibilityScanner.instance?.findNodeById(action.nodeId)
            if (node != null) {
                Log.d(TAG, "[Click] Node found, performing ACTION_CLICK")
                val result = node.performAction(android.view.accessibility.AccessibilityNodeInfo.ACTION_CLICK)
                Log.d(TAG, "[Click] Node click result: $result")
                return result
            }
            Log.w(TAG, "[Click] Node ${action.nodeId} not found, falling back to coordinate click")
        }

        action.bounds?.let { bounds ->
            Log.d(TAG, "[Click] Executing coordinate click at (${bounds.centerX()}, ${bounds.centerY()})")
            val result = actionExecutor.click(bounds.centerX(), bounds.centerY())
            Log.d(TAG, "[Click] Coordinate click result: $result")
            result
        } ?: run {
            Log.e(TAG, "[Click] No nodeId or bounds available, click failed")
            false
        }
    } catch (e: Exception) {
        Log.e(TAG, "[Click] Exception during click: ${e.message}", e)
        false
    }
}

// 打开应用日志

            }
        }

        result
    } catch (e: Exception) {
        Log.e(TAG, "[Navigate] Exception during navigation: ${e.message}", e)
        false
    }
```

```kotlin
class CloudAgent(...) {
    companion object {
        private const val TAG = "CloudAgent"
    }

    fun startTask(goal: String): ReActSession {
        Log.d(TAG, "Starting new task: $goal")
        // ...
    }

    suspend fun step(session: ReActSession, observation: String? = null): ReActResult {
        Log.d(TAG, "Executing step for task: ${session.taskId}")

        observation?.let {
            Log.d(TAG, "Adding observation: $it")
            session.conversationHistory.add(ReActMessage(role = "observation", content = it))
        }

        // 检查无限循环
        if (session.isInfiniteLoop()) {
            Log.w(TAG, "Detected infinite loop")
            return ReActResult(...)
        }

        // 检查超时
        if (System.currentTimeMillis() - session.startTime > 10 * 60 * 1000) {
            Log.w(TAG, "Task timeout")
            session.status = SessionStatus.TIMEOUT
            return ReActResult(...)
        }

        // 构建上下文
        val currentUiTree = AccessibilityScanner.getUiTreeSync()
        val contextMessage = buildContextMessage(currentUiTree)

        // 调用 LLM
        Log.d(TAG, "Calling LLM with model: $model")
        val response = callLlm(session.conversationHistory, contextMessage)
        Log.d(TAG, "LLM response: $response")

        // 解析响应
        val result = parseReActResponse(response, currentUiTree)
        Log.d(TAG, "Parsed result - thought: ${result.thought}, action: ${result.action}, isComplete: ${result.isComplete}")

        // 记录动作
        result.action?.let { action ->
            Log.d(TAG, "Recording action: ${action::class.java.simpleName}")
            session.executedActions.add(ActionRecord(action = action, ...))
        }

        if (result.isComplete) {
            session.status = SessionStatus.COMPLETED
        }

        return result
    }

    private suspend fun callLlm(history: List<ReActMessage>, context: String): String {
        Log.d(TAG, "callLlm: Building request with ${history.size} messages")

        val messages = history.map { ... } + Message(role = "user", content = context)

        val request = ChatRequest(model = model, messages = messages, ...)

        try {
            Log.d(TAG, "callLlm: Sending request to LLM")
            val response = llmClient.chatCompletion(request)
            val content = response.choices.firstOrNull()?.message?.content ?: ""
            Log.d(TAG, "callLlm: Received response with ${content.length} characters")
            return content
        } catch (e: Exception) {
            Log.e(TAG, "callLlm: Failed to get LLM response: ${e.message}", e)
            throw e
        }
    }
}
```

```kotlin
class AccessibilityActionExecutor(...) : ActionExecutor {
    companion object {
        private const val TAG = "ActionExecutor"
    }

    override suspend fun click(x: Int, y: Int): Boolean {
        Log.d(TAG, "[click] Executing click at ($x, $y)")
        val result = performGesture { ... }
        Log.d(TAG, "[click] Click result: $result")
        return result
    }

    override suspend fun swipe(fromX: Int, fromY: Int, toX: Int, toY: Int, durationMs: Long): Boolean {
        Log.d(TAG, "[swipe] Executing swipe from ($fromX, $fromY) to ($toX, $toY) in ${durationMs}ms")
        val result = performGesture { ... }
        Log.d(TAG, "[swipe] Swipe result: $result")
        return result
    }

    override suspend fun type(text: String): Boolean {
        Log.d(TAG, "[type] Executing type: \"$text\"")
        val currentNode = service.rootInActiveWindow ?: run {
            Log.e(TAG, "[type] Failed: rootInActiveWindow is null")
            return false
        }

        val editableNode = findEditableNode(currentNode)
        if (editableNode == null) {
            Log.e(TAG, "[type] Failed: no editable node found")
            return false
        }

        Log.d(TAG, "[type] Found editable node: ${editableNode.className}")

        val arguments = android.os.Bundle().apply {
            putCharSequence(AccessibilityNodeInfo.ACTION_ARGUMENT_SET_TEXT_CHARSEQUENCE, text)
        }

        val result = editableNode.performAction(AccessibilityNodeInfo.ACTION_SET_TEXT, arguments)
        Log.d(TAG, "[type] Type result: $result")
        return result
    }

    override suspend fun openApp(packageName: String): Boolean {
        Log.d(TAG, "[openApp] Executing open app: $packageName")
        return try {
            val intent = android.content.Intent().apply {
                action = android.content.Intent.ACTION_MAIN
                addCategory(android.content.Intent.CATEGORY_LAUNCHER)
                setPackage(packageName)
            }
            intent.addFlags(android.content.Intent.FLAG_ACTIVITY_NEW_TASK)
            service.startActivity(intent)
            Log.d(TAG, "[openApp] Started activity for: $packageName")
            true
        } catch (e: Exception) {
            Log.e(TAG, "[openApp] Failed to open app: ${e.message}", e)
            false
        }
    }

    override suspend fun navigateBack(): Boolean {
        Log.d(TAG, "[navigateBack] Executing back navigation")
        val result = service.performGlobalAction(AccessibilityService.GLOBAL_ACTION_BACK)
        Log.d(TAG, "[navigateBack] Result: $result")
        return result
    }

    override suspend fun goHome(): Boolean {
        Log.d(TAG, "[goHome] Executing home navigation")
        val result = service.performGlobalAction(AccessibilityService.GLOBAL_ACTION_HOME)
        Log.d(TAG, "[goHome] Result: $result")
        return result
    }

    override suspend fun openRecentApps(): Boolean {
        Log.d(TAG, "[openRecentApps] Executing recent apps navigation")
        val result = service.performGlobalAction(AccessibilityService.GLOBAL_ACTION_RECENTS)
        Log.d(TAG, "[openRecentApps] Result: $result")
        return result
    }
```

### 问题 2: 动作执行成功但无实际效果（用户反馈）

**现象**:
- 日志显示"执行成功"
- 但手机没有任何反应
- 例如："打开微信"说成功但没打开

**可能原因**:
1. `AccessibilityNodeInfo` 找到了但 `performAction` 实际失败（返回 true 但没执行）
2. 坐标计算错误（点击了错误位置）
3. 应用包名错误（Intent 启动失败但捕获异常返回 true）
4. 无障碍服务焦点问题（没有正确聚焦到目标应用）

**调试方法**:

### 重试机制
```kotlin
private suspend fun <T> executeWithRetry(
    operation: suspend (attempt: Int) -> T
): T {
    var lastException: Exception? = null
    var delayMs = BASE_DELAY_MS

    for (attempt in 0 until MAX_RETRIES) {
        try {
            return operation(attempt)
        } catch (e: RateLimitException) {
            lastException = e
            if (attempt < MAX_RETRIES - 1) {
                Log.w(TAG, "Rate limited, retrying in ${delayMs}ms...")
                kotlinx.coroutines.delay(delayMs)
                delayMs *= 2 // 指数退避
            }
        } catch (e: ServerException) {
            lastException = e
            if (attempt < MAX_RETRIES - 1) {
                Log.w(TAG, "Server error, retrying in ${delayMs}ms...")
                kotlinx.coroutines.delay(delayMs)
                delayMs *= 2
            }
        } catch (e: ApiException) {
            // API 错误不重试（认证错误、参数错误等）
            throw e
        }
    }

    throw lastException ?: ApiException("Unknown error during request")
}
```
